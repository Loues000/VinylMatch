<script>
    async function loadPlaylist() {
        // Backend liefert: { name, coverUrl, tracks: [...] }
        const response = await fetch("http://localhost:8888/api/playlist");
        const playlist = await response.json();

        // === Playlist-Header anzeigen ===
        const header = document.getElementById("playlist-header");
        header.innerHTML = `
        <img src="${playlist.playlistCoverUrl}" alt="Playlist Cover">
        <h2>${playlist.playlistName}</h2>
      `;

        // === Tracks in die Liste packen ===
        const container = document.getElementById("playlist");
        container.innerHTML = ""; // vorherigen "Loading..." Text entfernen

        playlist.tracks.forEach(track => {
            const div = document.createElement("div");
            div.className = "track";

            const img = document.createElement("img");
            img.src = track.coverUrl;
            img.className = "cover";

            const infoDiv = document.createElement("div");
            infoDiv.className = "info";

            const albumDiv = document.createElement("div");
            albumDiv.className = "album";
            albumDiv.textContent = track.album;

            const songDiv = document.createElement("div");
            songDiv.className = "song";
            songDiv.innerHTML = `${track.trackName} – <em>${track.artist}</em>`;

            infoDiv.appendChild(albumDiv);
            infoDiv.appendChild(songDiv);

            div.appendChild(img);
            div.appendChild(infoDiv);

            container.appendChild(div);
        });
    }

    loadPlaylist();
</script>

type TrackData = {
    trackName: string;
    artist: string;
    album: string;
    coverUrl: string
};

type PlaylistData ={
    playlistName: string;
    playlistCoverUrl: string;
    tracks: TrackData[];
};

async function loadPlaylist() {
    const response = await fetch("http://localhost:8888/api/playlist");
    const playlist: PlaylistData = await response.json();

        // Playlist-Header anzeigen
    const header = document.getElementById("playlist-header")!;
    header.innerHTML = `
      <img id="playlist-cover" src="${playlist.playlistCoverUrl}">
      <div class="header-info">
        <div class="playlist-title">${playlist.playlistName}</div>
        <div class="playlist-subtitle">${playlist.tracks.length} Songs</div>
      </div>
    `;

    // Jetzt die Farbe analysieren, sobald das Bild geladen ist
        const img = document.getElementById("playlist-cover") as HTMLImageElement;

    // Wir warten, bis das Bild geladen ist (wichtig!)
        img.addEventListener("load", () => {
            const colorThief = new (window as any).ColorThief();

        // Dominante Farbe + Farbpalette
        const dominant: number[] = colorThief.getColor(img);         // z.B. [123, 45, 67]
        const palette: number[][] = colorThief.getPalette(img, 2);   // 2 Farben für Verlauf

        // Fallback falls keine Palette
        const color1 = palette[0] ?? dominant;
        const color2 = palette[1] ?? dominant;

        // RGB → CSS-String
        const rgb1 = `rgb(${color1.join(",")})`;
        const rgb2 = `rgb(${color2.join(",")})`;

        // Hintergrund als Farbverlauf setzen
        document.body.style.background = `linear-gradient(135deg, ${rgb1}, ${rgb2})`;
    });

    const container = document.getElementById("playlist")!;
    container.innerHTML = `
        <h2>${playlist.playlistName}</h2>
        <img src="${playlist.playlistCoverUrl}" style ="width:200px;border-radius: 12px;">
    `;

    playlist.tracks.forEach(track => {
        const div = document.createElement("div");
        div.className = "track";

        const img =document.createElement("img");
        img.src = track.coverUrl;
        img.className = "cover";

        const infoDiv =document.createElement("div");
        infoDiv.className = "info";

        const albumDiv = document.createElement("div");
        albumDiv.className = "album";
        albumDiv.textContent = track.album;

        const songDiv = document.createElement("div");
        songDiv.className = "song";
        songDiv.innerHTML = `${track.trackName} – <em>${track.artist}</em>`;

        infoDiv.appendChild(albumDiv);
        infoDiv.appendChild(songDiv);

        div.appendChild(img);
        div.appendChild(infoDiv)

        container.appendChild(div);
    });
}

loadPlaylist();
